💻 命令行交互示例 可执行
# 1. 首先设置JWT（模拟）
export ADMIN_JWT="eyJhbGciOiJSUzI1NiIsInR5cCI6IkpXVCJ9..."
# 2. 列出当前租户所有策略
curl -X GET http://pep-proxy:8000/api/v1/policies \
-H "Authorization: Bearer $ADMIN_JWT" \
-H "X-Tenant-ID: tenant-1"
# 2. 查看可用模板
curl -X GET http://pep-proxy:8000/api/v1/policies/templates \
-H "Authorization: Bearer $ADMIN_JWT"
# 3. 预览模板效果
curl -X POST http://pep-proxy:8000/api/v1/policies/template/role_based \
-H "Authorization: Bearer $ADMIN_JWT" \
-H "Content-Type: application/json" \
-d '{ }'
# 4. 创建策略
curl -X POST http://pep-proxy:8000/api/v1/policies/role \
-H "Authorization: Bearer $ADMIN_JWT" \
-H "Content-Type: application/json" \
-d '{ }'
# 5. 更新策略 (PUT)
curl -X PUT http://pep-proxy:8000/api/v1/policies/role/viewer_documents_read \
-H "Authorization: Bearer $ADMIN_JWT" \
-H "Content-Type: application/json" \
-d '{"actions": ["read", "list"]}'
# 6. 删除策略 (DELETE)
curl -X DELETE http://pep-proxy:8000/api/v1/policies/role/viewer_documents_read \
-H "Authorization: Bearer $ADMIN_JWT"
